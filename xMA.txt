string xMaTip   = input.string(defval="(T)EMA", options = ["EMA", "(D)EMA", "(T)EMA", "SMA", "HMA", "RMA", "WMA"], title=  "xMA tipi", group="xMA Değerleri")

demaHesapla(src,len)=>
    ema1 = ta.ema(src, len)
    ema2 = ta.ema(ema1, len)
    (2 * ema1) - ema2  

temaHesapla(src,len)=>
    ema1 = ta.ema(src, len)
    ema2 = ta.ema(ema1, len)
    ema3 = ta.ema(ema2, len)
    3 * (ema1 - ema2) + ema3  

xMaKontrol(xMaTip, src, len)=>
    float xMA = switch xMaTip
        "EMA" => ta.ema(src, len) 
        "SMA" => ta.sma(src, len) 
        "HMA" => ta.hma(src, len) 
        "RMA" => ta.rma(src, len) 
        "WMA" => ta.wma(src, len) 
        "(D)EMA" => demaHesapla(src,len)
        "(T)EMA" => temaHesapla(src,len)
        => 0.0
